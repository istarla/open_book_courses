# Media

HTML5 introduced native support for audio and video elements, eliminating the need for plugins like Flash. These elements provide a standardized way to embed multimedia content with built-in controls and extensive JavaScript APIs for custom functionality.

## Media Controls

Both audio and video elements come with built-in controls, but you can create custom interfaces for enhanced user experience and branding consistency.

### Custom Video Player

```html
<div class="custom-video-player">
  <video id="customVideo" width="640" height="360" poster="video-poster.jpg">
    <source src="sample-video.mp4" type="video/mp4">

  </video>
  
  <div class="video-controls">
    <button id="videoPlayPause">‚ñ∂Ô∏è</button>

    <input type="range" id="videoVolume" min="0" max="100" value="100">

<script>
const video = document.getElementById('customVideo');
const playPauseBtn = document.getElementById('videoPlayPause');
const progressBar = document.getElementById('videoProgress');
const currentTimeSpan = document.getElementById('videoCurrentTime');
const durationSpan = document.getElementById('videoDuration');
const muteButton = document.getElementById('muteButton');
const volumeSlider = document.getElementById('videoVolume');
const fullscreenBtn = document.getElementById('fullscreenButton');

// Play/Pause functionality
playPauseBtn.addEventListener('click', () => {
  if (video.paused) {
    video.play();
    playPauseBtn.textContent = '‚è∏Ô∏è';
  } else {
    video.pause();
    playPauseBtn.textContent = '‚ñ∂Ô∏è';
  }
});

// Update progress bar and time
video.addEventListener('timeupdate', () => {
  const progress = (video.currentTime / video.duration) * 100;
  progressBar.value = progress;
  currentTimeSpan.textContent = formatTime(video.currentTime);
});

// Seek functionality
progressBar.addEventListener('input', () => {
  const seekTime = (progressBar.value / 100) * video.duration;
  video.currentTime = seekTime;
});

// Mute/unmute functionality
muteButton.addEventListener('click', () => {
  if (video.muted) {
    video.muted = false;
    muteButton.textContent = 'üîä';
  } else {
    video.muted = true;
    muteButton.textContent = 'üîá';
  }
});

// Volume control
volumeSlider.addEventListener('input', () => {
  video.volume = volumeSlider.value / 100;
  if (video.volume === 0) {
    muteButton.textContent = 'üîá';
  } else {
    muteButton.textContent = 'üîä';
  }
});

// Fullscreen functionality
fullscreenBtn.addEventListener('click', () => {
  if (video.requestFullscreen) {
    video.requestFullscreen();
  } else if (video.webkitRequestFullscreen) {
    video.webkitRequestFullscreen();
  } else if (video.mozRequestFullScreen) {
    video.mozRequestFullScreen();
  }
});

// Display duration when metadata loads
video.addEventListener('loadedmetadata', () => {
  durationSpan.textContent = formatTime(video.duration);
});

// Reset play button when video ends
video.addEventListener('ended', () => {
  playPauseBtn.textContent = '‚ñ∂Ô∏è';
});

// Format time helper function
function formatTime(seconds) {
  const minutes = Math.floor(seconds / 60);
  const secs = Math.floor(seconds % 60);
  return `${minutes}:${secs.toString().padStart(2, '0')}`;
}
</script>
```

### Media Events and JavaScript API

```html
<div class="media-player-advanced">
  <video id="advancedVideo" width="600" height="338" controls>
    <source src="demo-video.mp4" type="video/mp4">
  </video>
  
  <div class="playback-controls">
    <button onclick="changePlaybackRate(0.5)">0.5x</button>
    <button onclick="changePlaybackRate(1)">1x</button>
    <button onclick="changePlaybackRate(1.25)">1.25x</button>
    <button onclick="changePlaybackRate(1.5)">1.5x</button>
    <button onclick="changePlaybackRate(2)">2x</button>
  </div>
  
  <div class="media-info" id="mediaInfo">
    <p>Status: <span id="videoStatus">Paused</span></p>
    <p>Current Time: <span id="currentTime">0:00</span></p>
    <p>Duration: <span id="duration">Loading...</span></p>
    <p>Playback Rate: <span id="playbackRate">1x</span></p>
    <p>Volume: <span id="volumeLevel">100%</span></p>
  </div>
</div>

<script>
const advancedVideo = document.getElementById('advancedVideo');

// Event listeners for video events
advancedVideo.addEventListener('loadstart', () => console.log('Started loading'));
advancedVideo.addEventListener('loadeddata', () => console.log('Data loaded'));
advancedVideo.addEventListener('canplay', () => console.log('Can start playing'));
advancedVideo.addEventListener('canplaythrough', () => console.log('Can play through'));

advancedVideo.addEventListener('play', () => {
  document.getElementById('videoStatus').textContent = 'Playing';
});

advancedVideo.addEventListener('pause', () => {
  document.getElementById('videoStatus').textContent = 'Paused';
});

advancedVideo.addEventListener('ended', () => {
  document.getElementById('videoStatus').textContent = 'Ended';
});

advancedVideo.addEventListener('timeupdate', () => {
  document.getElementById('currentTime').textContent = formatTime(advancedVideo.currentTime);
});

advancedVideo.addEventListener('loadedmetadata', () => {
  document.getElementById('duration').textContent = formatTime(advancedVideo.duration);
});

advancedVideo.addEventListener('volumechange', () => {
  const volumePercent = Math.round(advancedVideo.volume * 100);
  document.getElementById('volumeLevel').textContent = `${volumePercent}%`;
});

// Playback rate control
function changePlaybackRate(rate) {
  advancedVideo.playbackRate = rate;
  document.getElementById('playbackRate').textContent = `${rate}x`;
}

// Keyboard controls
document.addEventListener('keydown', (e) => {
  if (document.activeElement === advancedVideo || 
      document.activeElement === document.body) {
    switch(e.key) {
      case ' ':  // Spacebar
        e.preventDefault();
        if (advancedVideo.paused) {
          advancedVideo.play();
        } else {
          advancedVideo.pause();
        }
        break;
      case 'ArrowLeft':  // Left arrow - seek backward
        e.preventDefault();
        advancedVideo.currentTime -= 10;
        break;
      case 'ArrowRight':  // Right arrow - seek forward
        e.preventDefault();
        advancedVideo.currentTime += 10;
        break;
      case 'ArrowUp':  // Up arrow - volume up
        e.preventDefault();
        advancedVideo.volume = Math.min(1, advancedVideo.volume + 0.1);
        break;
      case 'ArrowDown':  // Down arrow - volume down
        e.preventDefault();
        advancedVideo.volume = Math.max(0, advancedVideo.volume - 0.1);
        break;
    }
  }
});

function formatTime(seconds) {
  const minutes = Math.floor(seconds / 60);
  const secs = Math.floor(seconds % 60);
  return `${minutes}:${secs.toString().padStart(2, '0')}`;
}
</script>
```

### Audio with Different Formats

Different browsers support different audio formats, so it's best to provide multiple sources:

```html
<h3>Podcast Episode</h3>
<audio controls preload="metadata">
  <source src="podcast-episode-1.mp3" type="audio/mpeg">
  <source src="podcast-episode-1.ogg" type="audio/ogg">
  <p>Your browser doesn't support HTML5 audio. 
     <a href="podcast-episode-1.mp3">Download the audio file</a>.
  </p>
</audio>

<h3>Background Music</h3>
<audio controls loop>
  <source src="background-music.webm" type="audio/webm">
  <source src="background-music.mp3" type="audio/mpeg">
  <p>Audio not supported in your browser.</p>
</audio>
```

### Audio Attributes

Important attributes for audio elements:

```html
<!-- Autoplay audio (use carefully due to browser policies) -->
<audio controls autoplay muted>
  <source src="intro.mp3" type="audio/mpeg">
</audio>

<!-- Preload audio metadata -->
<audio controls preload="metadata">
  <source src="song.mp3" type="audio/mpeg">
</audio>

<!-- Loop audio continuously -->
<audio controls loop>
  <source src="ambient.mp3" type="audio/mpeg">
</audio>
```

## Embedding Video

The `<video>` element provides native video playback capabilities with extensive control options and accessibility features.

### Basic Video Implementation

```html
<video width="640" height="360" controls>
  <source src="movie.mp4" type="video/mp4">
  <source src="movie.webm" type="video/webm">
  <source src="movie.ogg" type="video/ogg">
  Your browser does not support the video tag.
</video>
```
